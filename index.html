func constructRectangle(area: Int) -> [Int] {
guard area > 0 else {
return [0,0]
}
var s = sqrt(Double(area))
while( area % Int(s) != 0 ){
s += 1;
}
return [area/Int(s) > Int(s) ? area/Int(s):Int(s),area/Int(s) > Int(s) ? Int(s):area/Int(s)]
}

new

基本的常量let，变量var，类型class，方法func，结构体struct，枚举enum，元组（）
let 定义常量,var 定义变量,let定义的必须在声明时指定初始值,普通的var声明也必须赋初始值.
let 声明的变量不可再改变,而var变量可以再改变,但是你不能再声明一个已经声明的常量或变量
可以在一行同时声明多个常量或变量,用逗号隔开;如果每个常量对应一个var或let只需要用分号隔开即可.
如果你想使用关键字来命名是不允许的,但是你可以在前后加上来命名,比如`let`,非关键字加命名也可以的.
可以用//进行注释
另外 我感觉这个好难 学起来很痛苦 慢慢来吧 先写这么多 以后慢慢写
